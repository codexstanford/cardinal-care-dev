covered(Policy,C) :-
    policy_covers_patient(P) &
    ~policy_opted_out(P) &
    tier_1_hospital(C) &
    ~exclusion_applies(C)

policy_opted_out(P):-
    policy.opted_out(P, yes)

policy_covers_patient(P):-
    policy.covers_patient(P, yes)

%% Armed forces
exclusion_applies(C):-
    claim.hospitalization(C, H) & 
    hospitalization.causal_event(H, military_service)

%% Alternative Healthcare
exclusion_applies(C):-
    claim.hospitalization(C, H) & 
    ~ hospitalization.type_of_healthcare(H, conventional_medicine)

%% Is tier 1
tier_1_hospital(C):-
    claim.hospitalization(C, H) & 
    hospitalization.hospital(H, Hos) &
    hospital.tier(Hos, 1)
